\begin{align}
	a &= b\\
	E &= mc^2 + \int_a^a x\, \mathrm{d}x
\end{align}
\end{theorem}

	\begin{empheq}[left=L\Rightarrow\Rrightarrow\empheqlbrace, box=\mymathbox, right=\empheqrbrack\Lleftarrow\Leftarrow R]{alignat=1}

	\end{empheq}
\end{proof}
